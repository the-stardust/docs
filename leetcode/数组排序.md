## 数组排序
### 快速排序
```go
func main(){
    arr := []int{1,2,5,3,4,11,21,18}
    quickSort(arr,0,len(arr) - 1)
    fmt.Println(arr)
}
func quickSort(nums []int,start,end int){
	if start < end {
		pivot := pivot(nums, 0, end)
		quickSort(nums,0,pivot - 1)
		quickSort(nums,pivot + 1,end)
	}
}
func pivot(nums []int,start,end int)int{
	pivot := nums[end]
	for i := start;i < end;i++{
		if nums[i] < pivot {
			nums[start],nums[i] = nums[i],nums[start]
			start++
		}
	}
	nums[start],nums[end] = nums[end],nums[start]
	return start
}
```
### 归并排序
```go
func main(){
    arr := []int{1,2,5,3,4,11,21,18}
    arr = mergeSort(arr)
    fmt.Println(arr)
}

func mergeSort(nums []int) []int{
	if len(nums) <= 1 {
		return nums
	}
	mid := len(nums) / 2

	left := mergeSort(nums[:mid])
	right := mergeSort(nums[mid:])

	return merge(left,right)
}

func merge(left,right []int)[]int{
	res := make([]int,len(left)+len(right))
	index := 0
	i,j := 0,0
	for i < len(left) && j < len(right) {
		if left[i] < right[j]{
			res[index] = left[i]
			i++
		}else{
			res[index] = right[j]
			j++
		}
		index++
	}
	for i < len(left) {
		res[index] = left[i]
		i++
		index++
	}
	for j < len(right) {
		res[index] = right[j]
		j++
		index++
	}
	return res
}
```

### 堆排序
```go
func main(){
    arr := []int{1,2,5,3,4,11,21,18}
    heapSort(arr)
    fmt.Println(arr)
}
func heapSort(nums []int){
	if len(nums) <= 1 {
		return
	}
	for i := len(nums) / 2 - 1;i >= 0;i--{
		heapify(nums,i,len(nums))
	}

	for i := len(nums)-1;i>= 1;i-- {
		nums[0],nums[i] = nums[i],nums[0]
		heapify(nums,0,i)
	}
}

func heapify(nums []int,root,length int){

	for {
		left_child := 2 * root + 1
		right_child := 2 * root + 2
		idx := root

		if left_child < length && nums[left_child] > nums[idx] {
			idx = left_child
		}

		if right_child < length && nums[right_child] > nums[idx] {
			idx = right_child
		}

		if idx == root {
			break
		}
		nums[root],nums[idx] = nums[idx],nums[root]
		root = idx
	}
}
```

### 归并排序
```go
func sortList(head *ListNode) *ListNode {
    //归并排序
    return mergeSort(head)
   
}
func mergeSort(head *ListNode)*ListNode{
    if head == nil || head.Next == nil {
        return head
    }
    // 分解为两个链表, head->mid  mid.Next -> end
    mid := findMid(head)
    tail := mid.Next
    mid.Next = nil

    left := mergeSort(head)
    right := mergeSort(tail)

    return merge(left,right)
}

func findMid(head *ListNode)*ListNode{
    slow,fast := head,head
    for fast.Next != nil && fast.Next.Next != nil {
        fast = fast.Next.Next
        slow = slow.Next
    }
    return slow
}

func merge(left,right *ListNode)*ListNode{
    dummy := &ListNode{}
    curr := dummy

    for left != nil && right != nil {
        if left.Val < right.Val {
            curr.Next = left
            left = left.Next
        }else{
            curr.Next = right
            right = right.Next
        }
        curr = curr.Next
    }

    if left != nil {
        curr.Next = left
    }
    if right != nil {
        curr.Next = right
    }

    return dummy.Next
}
```
